10  public class SendEmail {
11    private File file;
12    private String to;
13    private String sub;
14    private String msg;
15  
16    public SendEmail(String fileName) {
17      file = new File(fileName);
18      readText();
19    }
20  
21    private void readText() {
22      BufferedReader bufferedReader = null;
23      try {
24        bufferedReader = new BufferedReader(new FileReader(file));
25        Pattern emailPattern = Pattern.compile("To:");
26        Pattern subjectPattern = Pattern.compile("Subject:");
27        String temp;
28        StringBuilder content = new StringBuilder();
29        while ((temp = bufferedReader.readLine()) != null) {
30          Matcher emailMatch = emailPattern.matcher(temp);
31          Matcher subjectMatch = subjectPattern.matcher(temp);
32          if (temp.contains("2017"))
33            continue;
34          if (emailMatch.find())
35            to = temp.substring(emailMatch.end()).trim();
36          else if (subjectMatch.find())
37            sub = temp.substring(subjectMatch.end()).trim();
38          else content.append(temp).append(System.lineSeparator());
39        }
40        msg = content.toString();
41      } catch (IOException e) {
42        e.printStackTrace();
43      } finally {
44        if (bufferedReader != null) {
45          try {
46            bufferedReader.close();
47          } catch (IOException e) {
48            e.printStackTrace();
49          }
50        }
51      }
52    }
53  
54    public void send() {
55      //Get properties object
56      Properties props = new Properties();
57      props.put("mail.smtp.host", "smtp.gmail.com");
58      props.put("mail.smtp.socketFactory.port", "465");
59      props.put("mail.smtp.socketFactory.class",
60              "javax.net.ssl.SSLSocketFactory");
61      props.put("mail.smtp.auth", "true");
62      props.put("mail.smtp.port", "465");
63      String from = "hanyuxinbnu@gmail.com";
64      String password = "070519961996yy";
65      //get Session
66      Session session = Session.getDefaultInstance(props,
67              new javax.mail.Authenticator() {
68                protected PasswordAuthentication getPasswordAuthentication() {
69                  return new PasswordAuthentication(from, password);
70                }
71              });
72      //compose message
73      try {
74        MimeMessage message = new MimeMessage(session);
75        message.addRecipient(Message.RecipientType.TO, new InternetAddress(to));
76        message.setSubject(sub);
77        message.setText(msg);
78        //send message
79        Transport.send(message);
80        System.out.println("message sent successfully");
81      } catch (MessagingException e) {
82        throw new RuntimeException(e);
83      }
84    }
