23    private boolean checkArgument(String[] args) {
24      if (args.length == 0) {
25        System.out.println("Please insert argument!");
26        printUsage();
27        return false;
28      }
29      Pattern emailPattern = Pattern.compile("\\w*\\W*\\w+(.txt)");
30      Pattern csvPattern = Pattern.compile("(Flight)\\d+(From)[A-Z][a-z]+(To)[A-Z][a-z]+(.csv)");
31      Pattern csvWrongPattern = Pattern.compile("(Flight)\\d+(.csv)");
32      Pattern dirPattern = Pattern.compile("\\w+");
33      StringBuilder lostInforamtion = new StringBuilder();
34      boolean isError = false;
35      for (int i = 0; i < args.length; i++) {
36        if (i == args.length - 1 && args[i].indexOf("-") != -1) {
37          printError(lostInforamtion.toString());
38          printUsage();
39          return false;
40        }
41        if (args[i].equals("--csv-file")) {
42          if (csvPattern.matcher(args[i + 1]).matches())
43            CsvName = args[++i];
44          else if (csvWrongPattern.matcher(args[i + 1]).matches())
45            printCsvError();
46          else {
47            lostInforamtion.append(args[i]).append(" ");
48            isError = true;
49          }
50        } else if (args[i].equals("--email-template")) {
51          if (emailPattern.matcher(args[i + 1]).matches())
52            template = args[++i];
53          else {
54            lostInforamtion.append(args[i]).append(" ");
55            isError = true;
56          }
57        } else if (args[i].equals("--event")) {
58          if (!args[i + 1].equals("arrival") && !args[i + 1].equals("departure")) {
59            lostInforamtion.append(args[i]).append(" ");
60            isError = true;
61          } else event = args[++i];
62        } else if (args[i].equals("--output-dir")) {
63          if (dirPattern.matcher(args[i + 1]).matches()) {
64            dir = args[++i];
65          } else {
66            lostInforamtion.append(args[i]).append(" ");
67            isError = true;
68          }
69        }
70      }
71      if (isError) {
72        printError(lostInforamtion.toString());
73        printUsage();
74        return false;
75      }
76      return true;
77    }
78  
79    private void printError(String error) {

81    }

85    }
