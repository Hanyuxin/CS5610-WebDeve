3   import java.io.BufferedReader;
4   import java.io.File;
5   import java.io.FileInputStream;
6   import java.io.IOException;
7   import java.io.InputStream;
8   import java.io.InputStreamReader;
9   import java.io.Reader;
10  
11  public class IOLibrary {
12  
13    public String readFile(File file) {
14      BufferedReader bufferedReader = null;
15      StringBuilder stringBuilder = new StringBuilder();
16      try {
17        InputStream inputStream = new FileInputStream(file);
18        Reader rder = new InputStreamReader(inputStream, "UTF-8");
19        bufferedReader = new BufferedReader(rder);
20        String temp;
21        while ((temp = bufferedReader.readLine()) != null) {
22          stringBuilder.append(temp).append(System.lineSeparator());
23        }
24      } catch (IOException e) {
25        e.printStackTrace();
26      } finally {
27        if (bufferedReader != null) {
28          try {
29            bufferedReader.close();
30          } catch (IOException e) {
31            e.printStackTrace();
32          }
33        }
34      }
35      return stringBuilder.toString();
36    }
37  }
