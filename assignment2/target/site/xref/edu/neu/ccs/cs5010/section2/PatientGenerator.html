3   import java.util.Random;

5   public class PatientGenerator implements Generator {
6     private Random random = new Random();
7     private long arriveTimeLow;
8     private long arriveTimeHigh;
9   
10    PatientGenerator(long low, long high) {
11      arriveTimeLow = low;
12      arriveTimeHigh = high;
13    }
14  
15    /**
16     * randomly generator new Patient and return Patient
17     * @return Patient
18     */
19    public Patient next() {
20      try {
21        Patient patient = Patient.class.newInstance();
22        patient.setType(0);
23        patient.setArriveTime(arriveTimeLow
24                + random.nextInt((int) (arriveTimeHigh - arriveTimeLow + 1)));
25        patient.setUrgeLevel(1 + random.nextInt(10));
26        patient.setTreatment(random.nextInt(180));
27        return patient;
28      } catch (InstantiationException | IllegalAccessException e) {
29        e.printStackTrace();
30      }
31      return null;
32    }
33  }
