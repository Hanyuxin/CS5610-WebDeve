3   import java.util.List;
4   import java.util.concurrent.BlockingQueue;
5   
6   public class HourDataProducer implements Runnable {
7       private BlockingQueue<int[]> hourQueue;
8       private List<String> input;
9   
10      public HourDataProducer(BlockingQueue<int[]> hourQueue, List<String> input) {
11          this.hourQueue = hourQueue;
12          this.input = input;
13      }
14  
15      /**
16       * When an object implementing interface <code>Runnable</code> is used
17       * to create a thread, starting the thread causes the object's
18       * <code>run</code> method to be called in that separately executing
19       * thread.
20       * <p>
21       * The general contract of the method <code>run</code> is that it may
22       * take any action whatsoever.
23       *
24       * @see Thread#run()
25       */
26      @Override
27      public void run() {
28  
29          int hourPos = CsvInfo.getHourPos();
30          int liftPos = CsvInfo.getLiftPos();
31          if (input.size() == 0) {
32              throw new RuntimeException("There is no content in input file");
33          }
34  
35          for (int i = 1; i < input.size(); i++) {
36              String[] strs = input.get(i).split(",");
37              int hourID = Integer.parseInt(strs[hourPos]) / 60 + 1;
38              int liftID = Integer.parseInt(strs[liftPos]);
39              try {
40                  hourQueue.put(new int[]{hourID, liftID});
41              } catch (InterruptedException e) {
42                  e.printStackTrace();
43              }
44          }
45  
46          try {
47              hourQueue.put(new int[]{});
48          } catch (InterruptedException e) {
49              e.printStackTrace();
50          }
51      }
52  }
