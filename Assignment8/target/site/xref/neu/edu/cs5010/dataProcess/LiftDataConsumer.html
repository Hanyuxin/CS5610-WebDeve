11  public class LiftDataConsumer implements Runnable{
12  
13      private BlockingQueue<Integer> liftQueue;
14      private Map<Integer, Lift> map;
15  
16      public LiftDataConsumer(BlockingQueue<Integer> liftQueue, Map<Integer, Lift> map) {
17          this.liftQueue = liftQueue;
18          this.map = map;
19      }
20  
21  
22      /**
23       * When an object implementing interface <code>Runnable</code> is used
24       * to create a thread, starting the thread causes the object's
25       * <code>run</code> method to be called in that separately executing
26       * thread.
27       * <p>
28       * The general contract of the method <code>run</code> is that it may
29       * take any action whatsoever.
30       *
31       * @see Thread#run()
32       */
33      @Override
34      public void run() {
35          while (true){
36              int liftID = 0;
37              try {
38                  liftID = liftQueue.take();
39              } catch (InterruptedException e) {
40                  e.printStackTrace();
41              }
42              if(liftID == -1) break;
43              map.putIfAbsent(liftID, new Lift(liftID));
44              map.get(liftID).increaseRidesCount();
45          }
46  
47      }
48  }
