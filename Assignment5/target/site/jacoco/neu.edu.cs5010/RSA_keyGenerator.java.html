package neu.edu.cs5010;

import java.math.BigInteger;
import java.util.Random;

public class RSA_keyGenerator {

    private BigInteger n;
    private BigInteger e;
    private BigInteger d;
    private Random random;
    private int bitLength = 32;

    public RSA_keyGenerator() {
        random = new Random();
        BigInteger p = BigInteger.probablePrime(bitLength, random);
        BigInteger q = BigInteger.probablePrime(bitLength,random);
        n = p.multiply(q);
        BigInteger thi = p.subtract(BigInteger.ONE).multiply(q.subtract(BigInteger.ONE));
        e = BigInteger.probablePrime(bitLength / 2, random);
        while (thi.gcd(e).compareTo(BigInteger.ONE) > 0 && e.compareTo(thi) < 0
                && n.gcd(e).compareTo(BigInteger.ONE) > 0 && e.compareTo(n) < 0) {
            e.add(BigInteger.ONE);
        }
        d = e.modInverse(thi);
    }

    /**
     * get the public key
     * @return public Key
     */
    public BigInteger[] getPublicKey() {
        return new BigInteger[]{e,n};
    }

    /**
     * get the private Key
     * @return private Key
     */
    public BigInteger[] getPrivateKey() {
        return new BigInteger[]{d,n};
    }
}
