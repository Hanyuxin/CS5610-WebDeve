9       public RSA_verification(RSA_signature signature, Client client) {

13      private boolean verify(RSA_signature signature, Client client) {
14          BigInteger[] publicKey = client.getRsAgenerator().getPublicKey();
15          BigInteger decode = signature.getSignature().modPow(publicKey[0],publicKey[1]);
16          BigInteger message = new BigInteger(Integer.toString(client.getMessage().getContent()));
17          return decode.compareTo(message) == 0;
18      }
19  
20      public boolean isVerified() {
21          return verified;
22      }
