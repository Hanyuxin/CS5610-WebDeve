8   public class SecureBankVerificationSimulator {
9   
10      private Random random;
11      private static int transitionNumber;
12      private String fileContent;
13      private PriorityQueue<Client> queue;
14      private Set<Integer> depositRej;
15      private Set<Integer> withdrawalRej;
16      private Map<BigInteger[],List<Integer>> map;
17      private List<List<Integer>> sameKey;
18      private Client[] clientArray;
19      private int clientUpperBound;
20      private int  verifications;
21      private double fraction;

24      public SecureBankVerificationSimulator(){
25          transitionNumber = 0;
26          random = new Random();
27          queue = new PriorityQueue<>(11,
28                  (a, b) -> (a.getTransitions() - b.getTransitions()));
29          depositRej = new HashSet<>();
30          withdrawalRej = new HashSet<>();
31          map = new HashMap<>();
32          sameKey = new ArrayList<>();
33  
34      }
35  
36      public static void main(String[] args) {
37          SecureBankVerificationSimulator simulator = new SecureBankVerificationSimulator();
38          simulator.checkArgument(args);
39          simulator.simulate();
40          simulator.summary();
41      }

43      /**
44       * Check whether the input arguments are vaild
45       * @param args input arguments
46       */
47      private void checkArgument(String[] args) {
48          if(args.length != 4) {
49              throw new IllegalArgumentException("Please input correct argument");
50          }
51  
52          clientUpperBound = Integer.parseInt(args[0]);
53          verifications = Integer.parseInt(args[1]);
54          fraction = Double.parseDouble(args[2]);
55          outputFileName = args[3];
56  
57          if(clientUpperBound > 50000) {
58              throw new IllegalArgumentException("Number of unique bank clients should be less than 50000");
59          } if(verifications > 10000) {
60              throw new IllegalArgumentException("Number of unique verifications should be less than 10000");
61          } if(fraction > 1 || fraction < 0) {
62              throw  new IllegalArgumentException("Fraction of invalid messages should be in the range [0-1]");
63          } if(outputFileName.length() == 0) {
64              throw new IllegalArgumentException("Output file name is not valid");
65          }
66      }
67  
68      private void simulate(){
69  
70          clientArray = new Client[clientUpperBound];
71          for(int i = 0; i < clientUpperBound; i++) {
72              Client client = new Client(clientUpperBound, fraction);
73              clientArray[i] = client;
74          }
75  
76          fileContent = "Transaction number, Date, Time, Client ID, Message, Digital signature, " +
77                  "Verified, Transactions status.";
78  
79          for (transitionNumber = 0; transitionNumber < verifications; transitionNumber++) {
80              Client client = clientArray[random.nextInt(clientUpperBound)];
81              client.setTransitions(client.getTransitions()+1);
82              RSA_signature signature = new RSA_signature(client);
83              RSA_verification rsa_verification = new RSA_verification(signature,client);
84              fileContent += toString(client,signature,rsa_verification);
85  
86              if (getTransactionsStatus(client,rsa_verification).equals("deposit rejected")) {
87                  depositRej.add(client.getClientID());
88              } else if (getTransactionsStatus(client,rsa_verification).equals("withdrawal rejected")) {
89                  withdrawalRej.add(client.getClientID());
90              }
91  
92              BigInteger[] publicKey = client.getRsAgenerator().getPublicKey();
93              if (!map.containsKey(publicKey)) {
94                  map.put(publicKey, new ArrayList<>());
95              }
96              if(!map.get(publicKey).contains(client.getClientID()))
97                  map.get(publicKey).add(client.getClientID());
98              if(map.get(publicKey).size()>1)
99                  sameKey.add(map.get(publicKey));
100 
101 
102             queue.offer(client);
103             if (queue.size() > 10) {
104                 queue.poll();
105             }
106 
107         }
108         IOLibrary.write(outputFileName,fileContent);
109     }

111     public void summary() {
112         System.out.println("***********************************");
113         System.out.println("The number of distinct transactions with different IDs, but the same public key:");
114         for (List<Integer> list : sameKey) {
115             System.out.println(list);
116         }
117 
118         System.out.println("Top ten unique users with the largest number of transactions during the simulation");
119         for (Client i : queue) {
120             System.out.println(i.getClientID());
121         }
122         System.out.println("List of all IDs with rejected deposit transactions");
123         for (int i : depositRej) {
124             System.out.println(i);
125         }
126         System.out.println("List of all IDs with rejected withdrawal transactions");
127         for (int i : withdrawalRej) {
128             System.out.println(i);
129         }
130     }

151     private String getTransactionsStatus(Client client, RSA_verification verification) {
152         int number = client.getMessage().getNumber();
153         if(client.getMessage().getTransactionType() == TransactionType.Deposit) {
154             if (!verification.isVerified()|| number > client.getDepositLimit()) {
155                 return "deposit rejected";
156             } else return "deposit accepted";
157         } else {
158             if(!verification.isVerified() || number > client.getWithdrawLimit()) return "withdrawal rejected";
159             else return "withdrawal accepted";
160         }
161 
162     }

164     public String toString(Client client, RSA_signature signature, RSA_verification verification) {
165         StringBuilder sb = new StringBuilder();
166         sb.append(transitionNumber).append(", ").append(getDate()).append(", ").append(getTime()).append(", ")
167                 .append(client.getClientID()).append(", ").append(client.getMessage().getContent()).append(", ")
168                 .append(signature.getSignature().toString()).append(", ").append(verification.isVerified())
169                 .append(", ").append(getTransactionsStatus(client,verification));
170         System.out.println(sb.toString());
171         return sb.toString();
172     }
